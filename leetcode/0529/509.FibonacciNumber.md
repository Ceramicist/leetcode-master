# SONGLIU
```Go
func fib(n int) int {
    a,b := 0,1
    for i:=0; i<n; i++{
        a,b = b,a+b
    }
    return a
}
```

# LL_Yang
```Python
class Solution:
    def fib(self, n: int) -> int:
        if n < 2:
            return n

        p, q, r = 0, 0, 1
        for i in range(2, n + 1):
            p, q = q, r
            r = p + q

        return r
```
